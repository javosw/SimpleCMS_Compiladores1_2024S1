una publicacion = una pagina web 
publicar contenido
	escribir un archivo html
	colocar dicho html en el servidor

manejadores de contenido
	generan paginas web 
		permiten agregar contenido a las paginas de forma grafica

objetivo
	crear un manejador de contenido
	permite crear contenido desde un cliente para un servidor

entidades
	servidor
	editor: es la persona que crea contenido para publicar
	lector: es quien lee las publicaciones
	
servidor 
	protocolo: http
	puerto: 80
	
sitio.xml
	lenguaje: xml
	contiene la informacion de un sitio
		nombre 
		paginas
	pagina index:
		es el nodo raiz de un sitio 
		puede anidar paginas

paginator
	sinonimo: nucleo
	se ejecuta dentro del servidor
	manipula archivos sitio.xml
	el id de una pagina debe ser distinto al de otras paginas en este u otros sitios
	acciones
		son mensajes desde enviados desde el editor
		se crean con los elementos
			<accion>
			<acciones>

editor-app
	plataforma: escritorio
	ventanas
		sesion
			es un login falso: no envia datos de autenticacion al servidor
			el usuario que ingrese se en <parametro nombre="USUARIO_CREACION">
		acciones
			vistas
				escritura: para escribir el lenguaje acciones
				errores: errores lexicos, errores sintacticos con linea, columna, lexema y simbolo
				servidor: respuestas desde el servidor en formato xml
		estadisticas
			vistas
				escritura: para escribir el lenguaje sqcms
				servidor: respuestas desde el servidor

					<atributo nombre="ALINEACION">

						[IZQUIERDA]

				<atributos>
					* <atributo nombre="ORIGEN"> [https://example.com/x.png]
					<atributo nombre="ALINEACION"> 
						[CENTRAR]
						[IZQUIERDA]
						[DERECHA]
						[JUSTIFICAR]

					* <atributo nombre="ANCHO"> [360]
			VIDEO -> traduccion: <video>
				<parametros>
					<parametro nombre="CLASE"> [VIDEO]
				<atributos>
					* <atributo nombre="ORIGEN"> [https://example.com/x.mp4]
					* <atributo nombre="ALTURA"> [240]
					* <atributo nombre="ANCHO"> [360]
			MENU -> traduccion: una o mas <a>
				<parametros>
					<parametro nombre="CLASE"> [MENU]
				<atributos>
					<atributo nombre="PADRE"> [idPagina]
					<atributo nombre="ETIQUETAS"> [ grupo1 | grupo2 | grupo3 ]
				para cada pagina anidada de la pagina [idPagina]
					si esta pagina tiene una de las etiquetas [ grupo1 | ... ] 
						esta pagina se agrega al menu
		<etiquetas>
			etiqueta: 
				es una palabra que agrega una pagina a <<un grupo de paginas con propiedades similares>>
			<accion nombre="MODIFICAR_PAGINA">
				cada etiqueta de dicha pagina es eliminada 
				y luego se le agrega cada etiqueta en <etiquetas>
	sqcms
		no es case sensitive

			consultar COMPONENTE <clase> "sitio1.pagina1";
				<clase>: TODOS | TITULO | MENU | ...

	%#%#%#%#%# SITIOS %#%#%#%#%#

	accion: {
		nombre: NUEVO_SITIO_WEB
		parametros: {
			* ID: [idSitio]
			USUARIO_CREACION: [idUsuario]
			FECHA_CREACION: [fecha]
			USUARIO_MODIFICACION: [idUsuario]
			FECHA_MODIFICACION: [fecha]
		}
	}
	accion: {
		nombre: BORRAR_SITIO_WEB
		parametros: {
			* ID: [idSitio]
		}
	}

	%#%#%#%#%# PAGINAS %#%#%#%#%#

	accion: {
		nombre: NUEVA_PAGINA
		parametros: {
			* ID: [idPagina]
			TITULO: [miTitulo]
			* SITIO: [idSitio]
			* PADRE: [idPagina]
			USUARIO_CREACION: [idUsuario]
			FECHA_CREACION: [fecha]
			USUARIO_MODIFICACION: [idUsuario]
			FECHA_MODIFICACION: [fecha]
		}
		etiquetas: {}
	}
	accion: {
		nombre: MODIFICAR_PAGINA
		parametros: {
			* ID: [idPagina]
			*? TITULO: [miTitulo]
		}
		*? etiquetas: { }
	}
	accion: {
		nombre: BORRAR_PAGINA
		parametros: {
			* ID: [idPagina]
		}
	}
